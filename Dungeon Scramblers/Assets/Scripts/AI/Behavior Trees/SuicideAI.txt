tree("Root")
	parallel
		repeat mute tree("Dead")
		repeat mute tree("Attack")
		repeat mute tree("Move")
		

tree("Attack")
	while SeePlayerAndSetTarget
		fallback
			while AttackInRange
				sequence
					AttackPlayer
					Wait(0.5)
					Die

tree("Dead")
	sequence
		IsHealthLessThan(0.1)
		Die

tree("Move")
	fallback
		while not AtStoppingDistance
			MoveToPlayer
		while AtStoppingDistance
			StopMoving
