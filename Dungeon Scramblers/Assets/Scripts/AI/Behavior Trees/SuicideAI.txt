tree("Root")
	parallel
		repeat mute tree("Dead")
		repeat mute tree("Attack")
		

tree("Attack")
	while SeePlayerAndSetTarget
		parallel
			repeat mute tree("Move")
			fallback
				while AttackInRange
					sequence
						AttackPlayer
						EnsureDeath
						Wait(1.0)
						Die

tree("Dead")
	sequence
		IsHealthLessThan(0.0)
		AttackPlayer
		EnsureDeath
		Wait(1.0)
		Die

tree("Move")
	fallback
		while not AtStoppingDistance
			MoveToPlayer
		while AtStoppingDistance
			StopMoving
